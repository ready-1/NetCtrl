{% trans "Server-side Validation" %}

{% trans "Always implement server-side validation in your Django views:" %}


def form_view(request):
    if request.method == 'POST':
        form = MyForm(request.POST)
        if form.is_valid():
            form.save()
            messages.success(request, _('Form submitted successfully!'))
            return redirect('success_url')
    else:
        form = MyForm()
    return render(request, 'template.html', {'form': form})
